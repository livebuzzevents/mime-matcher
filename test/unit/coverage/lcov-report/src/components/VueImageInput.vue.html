<template>
  <div class="vii" @click="click">
    <img class="vii_img" ref="img" :alt="alt" v-if="src" :src="src">
    <figure class="vii_fallback" :aria-label="alt" v-if="!src">
      <slot></slot>
    </figure>
    <input ref="input" type="file" :name="name" :accept="accept" @change="change">
  </div>
</template>
 
<script>
import isNumeric from 'isnumeric'
import isType from '@/isMimeType/isMimeType'
 
export default {
  name: 'vue-image-input',
  props: {
    value: {
      type: String
    },
    alt: {
      type: String
    },
    accept: {
      type: String,
      default: 'image/*'
    },
    default: {
      type: String
    },
    name: {
      type: String
    },
    maxSize: {
      type: String,
      validator(maxSize) {
        if (isNumeric(maxSize)) {
          return true
        } else {
          console.error('Prop "maxSize" must be numeric.')
          return false
        }
      }
    }
  },
  computed: {
    src() {
      return this.value ? this.value : this.default
    }
  },
  methods: {
    click() {
      this.$emit('click')
      this.$refs.input.click()
    },
    validate(file) {
      if (this.maxSize) {
        if (file.size > this.maxSize) {
          this.$emit('error', {
            type: 'maxSize',
            expected: this.maxSize,
            actual: file.size
          })
          return false
        }
      }
      if (!isType(file.type).anyOf(this.accept)) {
        this.$emit('error', {
          type: 'fileType',
          expected: this.accept,
          actual: file.type
        })
        return false
      }
      return true
    },
    change() {
      const [file] = this.$refs.input.files
      if (file && this.validate(file)) {
        this.$emit('change', {
          file
        })
        this.reader.readAsDataURL(file)
      }
    }
  },
  mounted() {
    this.reader = new FileReader()
    this.reader.addEventListener('load', () => {
      this.$emit('input', this.reader.result)
    }, false)
  }
}
</script>
