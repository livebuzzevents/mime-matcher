const MIME_TYPE_REGEX = /^(\*|[a-z0-9._\-]+)\/(\*|[a-z0-9._\-]+)$/
 
function createMatcher(expected) {
  if (expected === '*') {
    return () => true
  } else {
    return (actual) => actual === expected
  }
}
 
function isValidMimeTypeString(mimeType) {
  return Boolean(mimeType && mimeType.match(MIME_TYPE_REGEX))
}
 
class MimeMatcher {
  expected = []
  constructor(...expected) {
    expected.forEach(mimeType => {
      if (isValidMimeTypeString(mimeType)) {
        const [type, subType] = mimeType.split('/')
        this.expected.push({
          typeMatcher: createMatcher(type),
          subTypeMatcher: createMatcher(subType)
        })
      } else {
        const msg = `Value "${mimeType}" is not valid mime type.It should have format "type/subtype".`
        throw new TypeError(msg)
      }
    })
 
  }
  match(actual) {
    if (isValidMimeTypeString(actual)) {
      const [actualType, actualSubType] = actual.split('/')
      return this.expected.some(({ typeMatcher, subTypeMatcher }) => {
        return typeMatcher(actualType) && subTypeMatcher(actualSubType)
      })
    } else {
      return false
    }
  }
}
 
export { isValidMimeTypeString }
 
export default MimeMatcher
